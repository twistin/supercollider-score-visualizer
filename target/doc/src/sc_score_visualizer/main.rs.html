use nannou::prelude::*;
use nannou_osc as osc;
use std::sync::mpsc::{self, Receiver};
use std::thread;
use std::collections::VecDeque;
use noise::{NoiseFn, Perlin};
use serde::Deserialize;
use std::fs;
use nannou_osc::Packet;

// =====================================================================
// üéµ CONFIGURACI√ìN Y MODELO DE DATOS
// =====================================================================

#[derive(Debug, Clone, Deserialize)]
struct Config {
    visual: VisualSettings,
    osc: OscSettings,
}

#[derive(Debug, Clone, Deserialize)]
struct VisualSettings {
    time_window: f64,
    max_events: usize,
    background_color: [u8; 3],
    event_fade_time: f64,
}

#[derive(Debug, Clone, Deserialize)]
struct OscSettings {
    port: u16,
}

impl Default for Config {
    fn default() -> Self {
        Self {
            visual: VisualSettings {
                time_window: 10.0,
                max_events: 200,
                background_color: [25, 28, 32], // Fondo oscuro por defecto
                event_fade_time: 3.0,
            },
            osc: OscSettings {
                port: 57124, // Puerto correcto por defecto
            },
        }
    }
}

#[derive(Debug, Clone)]
struct MusicalEvent {
    event_type: EventType,
    start_time: f64,
    duration: f64,
    amplitude: f32,
    // Atributos est√©ticos
    color: Hsv,
    shape_seed: f64,
}

#[derive(Debug, Clone)]
enum EventType {
    Point { freq: f32 },
    Glissando { start_freq: f32, end_freq: f32, curvature: f32 },
    Cluster { center_freq: f32, spread: f32, voices: u32 },
    Noise { center_freq: f32, bandwidth: f32 },
    SoundMass { components: Vec<(f32, f32)> },
}

struct Model {
    config: Config,
    events: VecDeque<MusicalEvent>,
    osc_receiver: Receiver<osc::Message>,
    app_start_time: f64,
    perlin: Perlin,
}

// =====================================================================
// üöÄ INICIALIZACI√ìN
// =====================================================================

fn main() {
    nannou::app(model).update(update).run();
}

fn model(app: &App) -> Model {
    // Cargar configuraci√≥n desde config.toml
    let config = match load_config() {
        Ok(cfg) => {
            println!("‚úÖ Configuraci√≥n 'config.toml' cargada exitosamente.");
            cfg
        }
        Err(e) => {
            eprintln!("‚ö†Ô∏è No se pudo cargar 'config.toml': {}. Usando configuraci√≥n por defecto.", e);
            Config::default()
        }
    };

    // Crear ventana con anti-aliasing (MSAA x8)
    app.new_window()
        .size(1600, 900)
        .view(view)
        .title("SC Score Visualizer [Est√©tica Profesional]")
        .msaa_samples(8)
        .build()
        .unwrap();

    // Configurar receptor OSC
    let (osc_sender, osc_receiver) = mpsc::channel();
    let osc_port = config.osc.port;

    thread::spawn(move || {
        let receiver = osc::receiver(osc_port).expect("No se pudo crear el receptor OSC");
        println!("üì° Receptor OSC activo en puerto {}", osc_port);
        for (packet, _addr) in receiver.iter() {
            match packet {
                Packet::Message(msg) => {
                    if osc_sender.send(msg).is_err() {
                        break;
                    }
                }
                Packet::Bundle(bundle) => {
                    for packet in bundle.content.into_iter() {
                        if let Packet::Message(msg) = packet {
                            if osc_sender.send(msg).is_err() {
                                break;
                            }
                        }
                    }
                }
            }
        }
    });

    Model {
        events: VecDeque::new(),
        app_start_time: app.time as f64,
        perlin: Perlin::new(rand::random()),
        config,
        osc_receiver,
    }
}

fn load_config() -> Result<Config, Box<dyn std::error::Error>> {
    let config_content = fs::read_to_string("config.toml")?;
    let config: Config = toml::from_str(&config_content)?;
    Ok(config)
}

// =====================================================================
// üîÑ L√ìGICA DE ACTUALIZACI√ìN
// =====================================================================

fn update(app: &App, model: &mut Model, _update: Update) {
    let current_time = app.time as f64 - model.app_start_time;

    // Procesar nuevos eventos OSC
    while let Ok(msg) = model.osc_receiver.try_recv() {
        if let Some(event) = parse_osc_message(&msg, current_time) {
            model.events.push_back(event);
        }
    }

    // Limpiar eventos viejos que ya terminaron de desvanecerse
    let fade_time = model.config.visual.event_fade_time;
    model.events.retain(|event| {
        let event_end_time = event.start_time + event.duration + fade_time;
        event_end_time > current_time
    });

    // Limitar el n√∫mero m√°ximo de eventos
    while model.events.len() > model.config.visual.max_events {
        model.events.pop_front();
    }
}

// =====================================================================
// üé® L√ìGICA DE DIBUJO (VIEW)
// =====================================================================

fn view(app: &App, model: &Model, frame: Frame) {
    let draw = app.draw();
    let win = app.window_rect();
    let current_time = app.time as f64 - model.app_start_time;

    // Fondo oscuro
    let bg = model.config.visual.background_color;
    draw.background().color(rgb8(bg[0], bg[1], bg[2]));

    // Dibujar cada evento
    for event in &model.events {
        render_event(&draw, &win, event, current_time, &model.config, &model.perlin);
    }

    draw.to_frame(app, &frame).unwrap();
}

// =====================================================================
// ‚ú® RENDERIZADO DE EVENTOS
// =====================================================================

fn render_event(draw: &Draw, win: &Rect, event: &MusicalEvent, current_time: f64, config: &Config, perlin: &Perlin) {
    let event_age = current_time - event.start_time;
    if event_age < 0.0 { return; }

    // Calcular alpha para aparici√≥n y desvanecimiento suave
    let fade_duration = config.visual.event_fade_time;
    let alpha = if event_age < event.duration {
        // Apareciendo y activo
        (event_age / (event.duration * 0.2).max(0.01)).min(1.0) // Fade in r√°pido
    } else {
        // Desvaneciendo
        1.0 - ((event_age - event.duration) / fade_duration).min(1.0) // Fade out
    } as f32;

    if alpha <= 0.0 { return; }

    let color_with_alpha = hsva(event.color.hue.into(), event.color.saturation, event.color.value, alpha);

    match &event.event_type {
        EventType::Point { freq } => {
            let x = time_to_x(event.start_time, current_time, win, config.visual.time_window);
            let y = freq_to_y(*freq, win);
            let radius = (event.amplitude * 20.0 + 2.0) * alpha;

            // Glow exterior
            draw.ellipse()
                .x_y(x, y)
                .radius(radius * 1.5)
                .color(hsva(event.color.hue.into(), event.color.saturation, event.color.value, alpha * 0.5));
            
            // N√∫cleo
            draw.ellipse()
                .x_y(x, y)
                .radius(radius)
                .color(color_with_alpha);
        }
        EventType::Glissando { start_freq, end_freq, curvature } => {
            let points = (0..=20).map(|i| {
                let t = i as f32 / 20.0; // Progreso a lo largo de la l√≠nea
                let time_offset = t as f64 * event.duration;
                
                let x = time_to_x(event.start_time + time_offset, current_time, win, config.visual.time_window);
                
                // Interpolar frecuencia con curvatura
                let t_curved = t.powf(1.0 + curvature * 2.0);
                let freq = *start_freq + (*end_freq - *start_freq) * t_curved;
                let y = freq_to_y(freq, win);
                
                pt2(x,y)
            });

            for point in points {
                draw.ellipse()
                    .xy(point)
                    .radius(event.amplitude * 4.0 + 1.0)
                    .color(color_with_alpha);
            }
        }
        EventType::Cluster { center_freq, spread, voices } => {
            let x = time_to_x(event.start_time, current_time, win, config.visual.time_window);
            for i in 0..*voices {
                let noise_val = perlin.get([event.shape_seed + i as f64 * 0.1, event_age * 0.5]) as f32;
                let freq_offset = (noise_val - 0.5) * 2.0 * spread;
                let y = freq_to_y(center_freq + freq_offset, win);
                let radius = (event.amplitude * 10.0 + 1.0) * alpha;

                draw.ellipse()
                    .x_y(x + noise_val * 20.0, y)
                    .radius(radius)
                    .color(color_with_alpha);
            }
        }
        // El renderizado de Noise y SoundMass puede ser m√°s complejo
        _ => {}
    }
}

fn parse_osc_message(msg: &osc::Message, current_time: f64) -> Option<MusicalEvent> {
    let (addr, args) = (msg.addr.as_str(), &msg.args);

    let event_type = match addr {
        "/event" if !args.is_empty() => {
            let event_name = args[0].clone().string()?;
            match event_name.as_str() {
                "point" if args.len() > 3 => Some(EventType::Point { freq: args[1].clone().float()? }),
                "gliss" if args.len() > 4 => Some(EventType::Glissando {
                    start_freq: args[1].clone().float()?,
                    end_freq: args[2].clone().float()?,
                    curvature: args[4].clone().float().unwrap_or(0.0),
                }),
                "cluster" if args.len() > 5 => Some(EventType::Cluster {
                    center_freq: args[1].clone().float()?,
                    spread: args[2].clone().float()?,
                    voices: args[3].clone().int()? as u32,
                }),
                "noise" if args.len() > 4 => Some(EventType::Noise {
                    center_freq: args[1].clone().float()?,
                    bandwidth: args[2].clone().float()?,
                }),
                "mass" if args.len() > 3 => {
                    let components: Vec<(f32, f32)> = args.iter().skip(4).collect::<Vec<_>>()
                        .chunks_exact(2)
                        .map(|chunk| (chunk[0].clone().float().unwrap_or(0.0), chunk[1].clone().float().unwrap_or(0.0)))
                        .collect();
                    Some(EventType::SoundMass { components })
                }
                _ => None,
            }
        }
        _ => None,
    }?;

    let amp_index = args.iter().position(|arg| matches!(*arg, osc::Type::Float(_)))? + 1;
    let dur_index = amp_index + 1;

    let amplitude = args.get(amp_index)?.clone().float()?;
    let duration = args.get(dur_index)?.clone().float()? as f64;
    
    let freq_for_color = match event_type {
        EventType::Point { freq } => freq,
        EventType::Glissando { start_freq, .. } => start_freq,
        EventType::Cluster { center_freq, .. } => center_freq,
        EventType::Noise { center_freq, .. } => center_freq,
        EventType::SoundMass { ref components } if !components.is_empty() => components[0].0,
        _ => 440.0,
    };

    Some(MusicalEvent {
        event_type,
        start_time: current_time,
        duration,
        amplitude,
        color: freq_to_color(freq_for_color),
        shape_seed: rand::random(),
    })
}

fn freq_to_y(freq: f32, win: &Rect) -> f32 {
    let min_freq_log = 20.0f32.log2();
    let max_freq_log = 20000.0f32.log2();
    let freq_log = freq.clamp(20.0, 20000.0).log2();
    map_range(freq_log, min_freq_log, max_freq_log, win.bottom(), win.top())
}

fn time_to_x(event_time: f64, current_time: f64, win: &Rect, time_window: f64) -> f32 {
    map_range(event_time, current_time - time_window, current_time, win.left(), win.right())
}

fn freq_to_color(freq: f32) -> Hsv {
    let min_log = 20.0f32.log10();
    let max_log = 20000.0f32.log10();
    let freq_log = freq.clamp(20.0, 20000.0).log10();
    let hue = map_range(freq_log, min_log, max_log, 0.5, 1.0); // Rango de azul a rojo
    hsv(hue, 0.8, 1.0) // Saturaci√≥n y brillo altos para colores vibrantes
}
